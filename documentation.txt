/noitamap { /* Noita Map Generator by Linkshot, from Hermit Dreams (2023) */

  ; Initialization
  /clear @biome_map /* First we clean the previous map */
  /window -p +b @biome_map 400 400 72 50 /* Now we reestablish its position (400,400) and size (72x50) */
  /drawrect -fr @biome_map 0 1 0 0 72 50 /* Fill the whole thing black in case user's default background is not black */
  var %noita.map.x 0 /* Declare the x coordinate we'll be drawing with */
  var %noita.map.y 0 /* Declare the y coordinate we'll be drawing with */
  var %noita.map.swapcount 0 /* Declare the starting amount of colour swaps */

  ; Colour Swap Method
  if (s isin $1) /* Check if user input "s" to specify this mode */
  { 
    /drawpic @biome_map 0 0 "C:\Users\Linkshot\AppData\LocalLow\Nolla_Games_Noita\data\biome_impl\biome_map.png" /* !! REPLACE WITH YOUR USER FOLDER FIRST !! Draws the main biome_map.png image from the unpacked data.wak to the window we made in Initialization */
    if (!$2) { var %noita.map.swapamount $rand(1,80) } /* If user did not specify an amount of swaps, pick a random number from 1 to 80 */
    else { var %noita.map.swapamount $2 } /* Checks the number after the block of flags in the input string */
    while (%noita.map.swapcount <= %noita.map.swapamount) /* Repeat until desired swaps have occurred */
    {
      var %noita.map.input1 $mid($read("C:\Users\Linkshot\Documents\Noita\Biome Map Colours.txt"),3,6) /* Grab the hex code from the biome colour list and remove the alpha. This is the biome we will be replacing. */
      var %noita.map.colour1 $mid(%noita.map.input1,5,2) $+ $mid(%noita.map.input1,3,2) $+ $mid(%noita.map.input1,1,2) /* Slice and swap the RGB order to BGR */
      var %noita.map.input2 $mid($read("C:\Users\Linkshot\Documents\Noita\Biome Map Colours.txt"),3,6) /* Do this process again. This is the biome we will be inserting. */
      var %noita.map.colour2 $mid(%noita.map.input2,5,2) $+ $mid(%noita.map.input2,3,2) $+ $mid(%noita.map.input2,1,2)
      /drawreplace -r @biome_map $base(%noita.map.colour1,16,10) $base(%noita.map.colour2,16,10) /* Convert from hexadecimal to Base 10 so the draw function parses correctly */
      inc %noita.map.swapcount 1 /* Confirm a swap has occurred */
    }
  } /* This overrides all other map-drawing methods currently */

  ; Brand New Map Method
  else {
    while (%noita.map.y <= 48) /* Cap y-axis to 48 tiles (Maybe adjustable later?) */
    {
      while (%noita.map.x <= 70) /* Cap x-axis to 70 tiles */
      {
        if (w isin $1) { /biomeweight } /* Run the weighted colour selection function if user specified the w flag */
        else /* Randomly pull from biome colour list by default */
        {
          var %noita.map.input $mid($read("C:\Users\Linkshot\Documents\Noita\Biome Map Colours.txt"),3,6) /* Grab the hex code from the biome colour list and remove the alpha. */
          var %noita.map.colour $mid(%noita.map.input,5,2) $+ $mid(%noita.map.input,3,2) $+ $mid(%noita.map.input,1,2) /* Slice and swap the RGB order to BGR */
        }
        if ($getdot(@biome_map,%noita.map.x,%noita.map.y) = 0) || ($rand(1,25) > 22) /* Check if tile has already been filled in. The biome list contains no black squares, and our background is black (#000000) by default. If we already drew to this tile, we add a 3/25 chance of drawing over it. */
        {
          if (c isin $1) /* Check if user wants clumps of biomes */
          {
            var %noita.map.xlong $rand(1,4) /* 25% chance to draw a horizontal line */
            var %noita.map.yhigh $rand(1,4) /* 25% chance to draw a vertical line */
            if ($calc(%noita.map.xlong + %noita.map.yhigh) = 2) /* If both passed, draw a rectangle */
            {
              var %noita.map.xlength $rand(2,15)
              var %noita.map.ylength $rand(2,15) /* Min size is 2x2, Max size is 15x15 */
              if ($calc(%noita.map.xlength + %noita.map.x) > 70) { dec %noita.map.xlength $calc(%noita.map.x + %noita.map.xlength - 70) }
              if ($calc(%noita.map.ylength + %noita.map.y) > 48) { dec %noita.map.ylength $calc(%noita.map.y + %noita.map.ylength - 48) } /* Cull the lengths of each axis to prevent spilling out of bounds */
              /drawrect -fr @biome_map $base(%noita.map.colour,16,10) 1 %noita.map.x %noita.map.y %noita.map.xlength %noita.map.ylength /* Draw a filled rectangle with the previously selected colour and dimensions. Converts from Base 16 to Base 10 so that it parses correctly. */
            }
            elseif (%noita.map.xlong = 1) /* Drawing just a horizontal line */
            {
              var %noita.map.xlength $rand(3,20) /* Set min,max length from 3 to 20 */
              if ($calc(%noita.map.xlength + %noita.map.x) > 70) { dec %noita.map.xlength $calc(%noita.map.x + %noita.map.xlength - 70) } /* Prevent spilling out of bounds */
              /drawline -r @biome_map $base(%noita.map.colour,16,10) 1 %noita.map.x %noita.map.y $calc(%noita.map.x + %noita.map.xlength - 1) %noita.map.y /* Get the end coordinate by comparing the current x-coordinate to the decided length */
            }
            elseif (%noita.map.yhigh = 1) /* Drawing just a vertical line */
            {
              var %noita.map.ylength $rand(3,48) /* Vertical can stretch all the way to the bottom if it wants, because the Cursed Rock boundaries do */
              if ($calc(%noita.map.ylength + %noita.map.y) > 48) { dec %noita.map.ylength $calc(%noita.map.y + %noita.map.ylength - 48) } /* Prevent spilling out of bounds */
              /drawline -r @biome_map $base(%noita.map.colour,16,10) 1 %noita.map.x %noita.map.y %noita.map.x $calc(%noita.map.y + %noita.map.ylength - 1) /* Get the end coordinate by comparing the current y-coordinate to the decided length */
            }
            else { /drawdot -r @biome_map $base(%noita.map.colour,16,10) 0 %noita.map.x %noita.map.y } /* If neither passed, simply draw a dot with size 0 at the current coordinates */
          }
          /drawdot -r @biome_map $base(%noita.map.colour,16,10) 0 %noita.map.x %noita.map.y /* Draw another dot for good measure (patches a bug with line drawing at edges) */
        }
        if (g isin $1) && (%noita.map.x = 35) && (%noita.map.y = 13) { /drawdot -r @biome_map $base(808020,16,10) 0 %noita.map.x %noita.map.y } /* If user specified the g flag, guarantee the game start tile (35,13) is mountain_left_entrance (#208080) */
        inc %noita.map.x 1 /* Confirm we have drawn at this coordinate and move right */
      }
      var %noita.map.x 0 /* Reset x-coordinate to far left */
      inc %noita.map.y 1 /* Move down one tile */
    }
  }

  ; Finish
  /drawsave -q100 @biome_map biome_map.bmp /* Save the finished map to a file */
  /run biome_map.bmp /* Open the file in an image editor */
}
